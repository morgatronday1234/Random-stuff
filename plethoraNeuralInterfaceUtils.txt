module = peripheral.wrap("back")

breakBlocks = false
power = 5
while(true) do
 event, key, held = os.pullEvent("key_up")
 
 if (key == keys.c) then
  playerData = module.getMetaOwner()
  module.fire(playerData.yaw, playerData.pitch, power, breakBlocks)
 elseif (key == keys.semicolon) then
  --Flip the state, Its a mess i know...
  
  if (breakBlocks == false) then
   breakBlocks = true
  elseif (breakBlocks == true) then
   breakBlocks = false
  end 
  
  print("Current break state: "..tostring(breakBlocks))
 elseif (key == keys.v) then
  playerData = module.getMetaOwner()
  module.launch(playerData.yaw, playerData.pitch, 4)
 end
end
